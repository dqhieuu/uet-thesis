\NewDocumentCommand{\includefigure}{O{} m}{
    \begin{figure}[H]
        \setlength{\abovecaptionskip}{15pt plus 3pt minus 2pt}
        \centering
        \vspace{1cm}
        \captionsetup{font=bf}
        \includegraphics[width=0.9\textwidth,height=0.9\textheight,keepaspectratio]{#2}
        \IfValueT{#1}{\caption{#1}}
        \label{fig:#2}
        \vspace{1cm}
    \end{figure}
}

\NewDocumentCommand{\includesmallfigure}{O{} m}{
    \begin{figure}[H]
        \setlength{\abovecaptionskip}{15pt plus 3pt minus 2pt}
        \centering
        \vspace{1cm}
        \captionsetup{font=bf}
        \includegraphics[width=0.6\textwidth,height=0.6\textheight,keepaspectratio]{#2}
        \IfValueT{#1}{\caption{#1}}
        \IfValueT{#2}{\label{fig:#2}}
        \vspace{1cm}
    \end{figure}
}

\NewDocumentCommand{\includesvgfigure}{O{} O{}}{
    \begin{figure}[H]
        \setlength{\abovecaptionskip}{15pt plus 3pt minus 2pt}
        \centering
        \vspace{1cm}
        \ifstrempty{#1}{
            \includesvg[width=0.9\textwidth,height=0.9\textheight,keepaspectratio]{#2}
        }{
            \includesvg[width=0.9\textwidth,height=0.9\textheight,keepaspectratio]{#2}
            \captionsetup{font=bf}
            \caption{#1}
            \label{fig:#2}
        }
    \end{figure}
}
